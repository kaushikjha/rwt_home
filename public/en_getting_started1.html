Getting started with Rwt

 

Also create the view composed by 'index.html.erb' and 'index.rb' in the app/view/desktop directory with the code:

  

<!DOCTYPE HTML PUBLIC "-//W3C//DTD HTML 4.01 Transitional//EN">
<html>
  <head>
    <title>Rwt - The Rails Web Toolkit</title>
    <meta http-equiv="Content-Type" content="text/html; charset=UTF-8">
    <%= rwt_javascript_libraries %>
    <script src="/desktop/index?format=js" type="text/javascript"></script>
  </head>
  <%= rwt_body %>
  </body>
</html>
and

rwt_app do
  toolbar do
    menu('Rwt') do
      menu_item('What is this?',call_view('/desktop/what'))
      menu_item('Getting started',call_view('/desktop/started'))
      menu_item('Explore',call_view('/desktop/explore'))
    end
    menu('Tests') do
      menu_item('First window',call_view('/test/first'))
    end
  end
end
